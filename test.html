function renderSalesChart() { const soldThroughD0 = parseInt(document.getElementById('sales-day1-total').value) || 2000000; const refundPct = parseFloat(document.getElementById('sales-refund-pct').value) || 8; const psBaseRatio = parseInt(document.getElementById('sales-ps-ratio').value) || 36; const xbBaseRatio = 10; const otherBaseRatio= 14; const stBaseRatio = Math.max(5, 100 - psBaseRatio - xbBaseRatio - otherBaseRatio); const lbl = document.getElementById('sales-steam-ratio-lbl'); if (lbl) lbl.textContent = stBaseRatio; const MILESTONE_DAY4 = 3000000; const PRE4_INCREMENT = Math.max(0, MILESTONE_DAY4 - soldThroughD0); const POST_D4_DECAY = 0.87; const DAILY_FLOOR = 20000; const OTHER_ALLOC = { ps: 0.30, st: 0.05, xb: 0.15, other: 0.50 }; const clamp = (v, min, max) => Math.max(min, Math.min(max, v)); // 순위 → 판매 탄성 함수 개선 (PS 0.35, Steam 0.40) const sig = (base, rank, exp) => { const b = Math.max(1, base || 1); const r = clamp(rank || b, 1, 80); return Math.pow(b / r, exp); }; // 기타 채널 비중 자연 증가 (최대 25%) const otherShareForDay = di => clamp(0.14 + di * 0.008, 0.14, 0.25); // PS 일별 가중 평균 순위 const psByDay = {}; historyData.forEach(h => { const di = getDayIndex(h.timestamp); if (di < 0) return; const avg = h.averages?.combined; if (avg == null || !isFinite(avg)) return; if (!psByDay[di]) psByDay[di] = { sum: 0, cnt: 0 }; psByDay[di].sum += avg; psByDay[di].cnt++; }); const psDayAvg = {}; Object.entries(psByDay).forEach(([d, v]) => { psDayAvg[parseInt(d)] = v.sum / v.cnt; }); // Steam 일별 가중 평균 순위 const stByDay = {}; stsHist.forEach(h => { const di = getDayIndex(h.timestamp); if (di < 0) return; let tw = 0, ts = 0; Object.entries(h.results || {}).forEach(([name, d]) => { const r = d?.rank; if (!r || r >= 100) return; const w = STS_W[name] || 1; ts += r * w; tw += w; }); if (!tw) return; if (!stByDay[di]) stByDay[di] = { sum: 0, cnt: 0 }; stByDay[di].sum += ts / tw; stByDay[di].cnt++; }); const stDayAvg = {}; Object.entries(stByDay).forEach(([d, v]) => { stDayAvg[parseInt(d)] = v.sum / v.cnt; }); const observedDays = new Set([...Object.keys(psDayAvg), ...Object.keys(stDayAvg)].map(Number)); const todayDay = Math.max(0, getDayIndex(Date.now())); const maxDay = Math.max(4, todayDay, observedDays.size ? Math.max(...observedDays) : 0); const days = Array.from({ length: maxDay + 1 }, (_, i) => i); const psBase = psDayAvg[Object.keys(psDayAvg)[0]] || 4.0; const stBase = stDayAvg[Object.keys(stDayAvg)[0]] || 3.0; const labels = []; const psRankSeries = [], stRankSeries = []; let psLastRank = psBase, stLastRank = stBase; days.forEach(di => { const date = new Date(LAUNCH_DATE.getTime() + di * 86400000); labels.push(`${date.getMonth() + 1}/${date.getDate()}`); const psR = psDayAvg[di] != null ? psDayAvg[di] : clamp(psLastRank * 1.05, 1, 80); const stR = stDayAvg[di] != null ? stDayAvg[di] : clamp(stLastRank * 1.06, 1, 80); psLastRank = psR; stLastRank = stR; psRankSeries.push(psR); stRankSeries.push(stR); }); // 개선된 신호 계산 (Steam 영향력 강화 + 탄성 상향) const psSignal = psRankSeries.map(r => (psBaseRatio / 100) * sig(psBase, r, 0.35)); const stSignal = stRankSeries.map(r => (stBaseRatio / 100) * sig(stBase, r, 0.40)); const xbSignal = days.map((_, i) => { const combinedBase = (psBase + stBase) / 2; const combinedRank = (psRankSeries[i] + stRankSeries[i]) / 2; return (xbBaseRatio / 100) * sig(combinedBase, combinedRank, 0.18); }); const comboSignal = days.map((_, i) => 0.45 * psSignal[i] + 0.55 * stSignal[i] // Steam 비중 상향 ); // 총 판매량 분배 const totalDaily = Array(days.length).fill(0); totalDaily[0] = soldThroughD0; // Day 1~4 분배 (300만 앵커) const day14Weights = []; for (let di = 1; di <= 4; di++) { const decay = Math.pow(0.89, di - 1); const momentum = clamp(comboSignal[di] / (comboSignal[0] || 1), 0.72, 1.18); day14Weights.push(Math.max(0.12, decay * momentum)); } const wSum = day14Weights.reduce((a, b) => a + b, 0) || 1; for (let di = 1; di <= 4; di++) { totalDaily[di] = Math.round(Math.max(180000, PRE4_INCREMENT * (day14Weights[di - 1] / wSum))); } // Day 5 이후 for (let di = 5; di < days.length; di++) { const signalRef = comboSignal[4] || comboSignal[0] || 1; const rel = clamp((comboSignal[di] || signalRef) / signalRef, 0.75, 1.20); totalDaily[di] = Math.round(Math.max(DAILY_FLOOR, totalDaily[4] * Math.pow(POST_D4_DECAY, di - 4) * rel)); } // 플랫폼별 분배 const psDaily = [], stDaily = [], xbDaily = [], otherDaily = []; days.forEach(di => { const gross = totalDaily[di] || 0; const otherPool = gross * otherShareForDay(di); const visiblePool = gross - otherPool; const rawPs = psSignal[di] || 0.01; const rawSt = stSignal[di] || 0.01; const rawXb = xbSignal[di] || 0.01; const rawSum = rawPs + rawSt + rawXb || 1; const psCore = visiblePool * rawPs / rawSum; const stCore = visiblePool * rawSt / rawSum; const xbCore = visiblePool * rawXb / rawSum; const psAdj = otherPool * OTHER_ALLOC.ps; const stAdj = otherPool * OTHER_ALLOC.st; const xbAdj = otherPool * OTHER_ALLOC.xb; let ps = Math.round(psCore + psAdj); let st = Math.round(stCore + stAdj); let xb = Math.round(xbCore + xbAdj); let other = Math.round(gross - ps - st - xb); if (other < 0) { xb = Math.max(0, xb + other); other = 0; } psDaily.push(ps); stDaily.push(st); xbDaily.push(xb); otherDaily.push(other); }); // 누적 계산 const psCum = [], stCum = [], xbCum = [], otherCum = [], totCum = []; let psAcc = 0, stAcc = 0, xbAcc = 0, otherAcc = 0; days.forEach((_, i) => { psAcc += psDaily[i]; stAcc += stDaily[i]; xbAcc += xbDaily[i]; otherAcc += otherDaily[i]; psCum.push(psAcc); stCum.push(stAcc); xbCum.push(xbAcc); otherCum.push(otherAcc); totCum.push(psAcc + stAcc + xbAcc + otherAcc); }); const todayIdx = Math.min(todayDay, totCum.length - 1); const totalGross = totCum[todayIdx] || totCum[totCum.length - 1] || 0; const psTotal = psCum[todayIdx] || psCum[psCum.length - 1] || 0; const stTotal = stCum[todayIdx] || stCum[stCum.length - 1] || 0; const xbTotal = xbCum[todayIdx] || xbCum[xbCum.length - 1] || 0; const otherTotal = otherCum[todayIdx] || otherCum[otherCum.length - 1] || 0; // 순보유량 (환불은 Steam + 기타 PC 중심) const refundableBase = stTotal + otherTotal * 0.65; const totalNet = Math.round(totalGross - refundableBase * (refundPct / 100)); // 요약 카드 업데이트 const cards = document.getElementById('sales-summary-cards'); if (cards) { cards.innerHTML = `
전체 누적 판매량
${fmtSales(totalGross)}
순보유량 추정
${fmtSales(totalNet)}
환불 ${refundPct}% 적용
PS5 누적
${fmtSales(psTotal)}
Steam 누적
${fmtSales(stTotal)}
`; } // 차트 렌더링 (기존 Chart.js 코드 유지) const chartOpts = { responsive: true, maintainAspectRatio: false, scales: { x: { ticks: { color: '#aaa' }, grid: { color: '#222' } }, y: { ticks: { color: '#aaa', callback: v => fmtSales(v) }, grid: { color: '#222' } } } }; // 일별 차트 const ctx1 = document.getElementById('salesDailyChart').getContext('2d'); if (salesDailyCI) salesDailyCI.destroy(); salesDailyCI = new Chart(ctx1, { type: 'bar', data: { labels: labels, datasets: [ { label: 'PS5', data: psDaily, backgroundColor: 'rgba(255,71,87,0.75)', stack: 'stack1' }, { label: 'Steam', data: stDaily, backgroundColor: 'rgba(74,158,255,0.75)', stack: 'stack1' }, { label: '기타', data: otherDaily, backgroundColor: 'rgba(155,89,182,0.7)', stack: 'stack1' } ] }, options: { ...chartOpts, scales: { ...chartOpts.scales, x: { stacked: true }, y: { stacked: true } } } }); // 누적 차트 const ctx2 = document.getElementById('salesCumChart').getContext('2d'); if (salesCumCI) salesCumCI.destroy(); salesCumCI = new Chart(ctx2, { type: 'line', data: { labels: labels, datasets: [ { label: '전체', data: totCum, borderColor: '#f1c40f', borderWidth: 3, tension: 0.3 }, { label: 'PS5', data: psCum, borderColor: '#ff4757', borderWidth: 2, tension: 0.3 }, { label: 'Steam',data: stCum, borderColor: '#4a9eff', borderWidth: 2, tension: 0.3 } ] }, options: chartOpts }); console.log(`✅ 판매량 모델 업데이트 완료 (Today Day ${todayDay}, 총 ${fmtSales(totalGross)})`); }
